# javaScript 注释

# 开发者信息注释

“ /** @author developer Name (Developer Email) */ ”

# 单行注释

“ // ” 后跟一个空格，缩进与下一行被注释说明代码一致。必须独占一行

# 文档化注释

以 “ /**...*/ ” 结尾，方便审阅代码和文档化代码

1. 函数 或 方法
2. 类
3. 类属性
4. 事件
5. 引入文件
6. nameSpace

推荐

// good

/**
 * 查询商品详情
 */
queryGoods() {
  // 所需执行流程
}

# 对象命名注释

“ /** nameSpace */ ”

//good

/**
 * @nameSpace
 */
 let data = {}

# 普通类注释

“ /** @class */ ”

// good

/**
 * 描述
 *
 * @class
 */
function handleDelete() {
  // 所需执行流程
}

# 继承类注释

“ /** @class @extends */ ”

// good

/**
 * 描述
 *
 * @class
 * @extends
 */
function handleLoading() {
  handleDelete()
}

util.inherits(handleDelete, handLoading)
